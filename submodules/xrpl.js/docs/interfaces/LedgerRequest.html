Interface LedgerRequest

Retrieve information about the public ledger. Expects a response in the form of a LedgerResponse.

example
const ledger: LedgerRequest = {
 "id": 14,
 "command": "ledger",
 "ledger_index": "validated",
 "full": false,
 "accounts": false,
 "transactions": false,
 "expand": false,
 "owner_funds": false
}
